$$
\begin{array}{ll}
\int_{-1}^{1} \frac{1}{2}(e^x + e^{-x}) \mathop{dx}
&= \frac{1}{2} \int_{-1}^{1} (e^x + e^{-x}) \mathop{dx} \\
&= \frac{1}{2} \Big|_{-1}^{1} (e^x - e^{-x}) \\
&= \frac{1}{2} [(e^1 - e^{(-1)}) - (e^{-1} - e^{-(-1)})] \\
&= \frac{1}{2} [e - e^{-1} - e^{-1} + e)] \\
&= \frac{1}{2} [2e - 2e^{-1}] \\
&= e - e^{-1} \\
\end{array}
$$
